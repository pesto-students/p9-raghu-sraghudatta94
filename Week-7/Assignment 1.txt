function subarray( numarray){
let currsum = 0;
let maxSum= -Infinity;

for(let i= 0;i<numarray.length;i++){

currsum= Math.max(numarray[i], currsum+numarray[i]);

maxSum= Math.max(currsum,maxSum);


}

return maxSum;
}
